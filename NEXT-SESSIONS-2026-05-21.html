-- Grandfather eligibility: 15-trace OR 7-distinct-day floor.
-- Capped at 100 unique cloud_user_ids. Frozen as of strategy lock date.
WITH eligible AS (
  SELECT u.id AS user_id
  FROM users u
  WHERE u.created_at < '2026-05-21'
    AND (
      (SELECT COUNT(*) FROM agent_traces WHERE user_id = u.id) >= 15
      OR (SELECT COUNT(DISTINCT date(created_at)) FROM execution_logs WHERE user_id = u.id) >= 7
    )
  ORDER BY u.created_at ASC
  LIMIT 100
)
INSERT INTO grandfather_cohort (user_id, qualified_at, legacy_rate_usd)
SELECT user_id, NOW(), 14.00 FROM eligible
ON CONFLICT DO NOTHING;
